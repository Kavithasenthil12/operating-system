#include <stdio.h>
#define MAX 10

int main() {
    int n, r;
    printf("Enter processes and resources: ");
    scanf("%d %d", &n, &r);

    int alloc[MAX][MAX], maxm[MAX][MAX];
    int avail[MAX], need[MAX][MAX];

    printf("Allocation Matrix:\n");
    for (int i=0;i<n;i++)
        for (int j=0;j<r;j++) scanf("%d",&alloc[i][j]);

    printf("Max Matrix:\n");
    for (int i=0;i<n;i++)
        for (int j=0;j<r;j++) scanf("%d",&maxm[i][j]);

    printf("Available:\n");
    for (int j=0;j<r;j++) scanf("%d",&avail[j]);

    for (int i=0;i<n;i++)
        for (int j=0;j<r;j++)
            need[i][j]=maxm[i][j]-alloc[i][j];

    int finish[MAX]={0}, safe[MAX], work[MAX];
    for (int j=0;j<r;j++) work[j]=avail[j];

    int count=0;
    while (count<n) {
        int found=0;
        for (int i=0;i<n;i++) {
            if (!finish[i]) {
                int ok=1;
                for (int j=0;j<r;j++)
                    if (need[i][j]>work[j]) { ok=0; break; }
                if (ok) {
                    for (int j=0;j<r;j++) work[j]+=alloc[i][j];
                    safe[count++]=i;
                    finish[i]=1;
                    found=1;
                }
            }
        }
        if (!found) {
            printf("DEADLOCK detected!\n");
            return 0;
        }
    }
    printf("System is SAFE.\nSafe Sequence: ");
    for (int i=0;i<n;i++)
        printf("P%d%s", safe[i], i<n-1?" -> ":"\n");
    return 0;
}
