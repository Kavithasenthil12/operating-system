#include <stdio.h>
#include <stdlib.h>
#include <sys/shm.h>
#include <string.h>

int main() {
    key_t key = 1234;
    int shmid = shmget(key, 1024, IPC_CREAT|0666);
    if (shmid<0) { perror("shmget"); exit(1); }

    char *shm = (char*)shmat(shmid, NULL, 0);
    printf("Enter message: ");
    fgets(shm, 1024, stdin);
    printf("Message written to shared memory.\n");
    shmdt(shm);
    return 0;
}

#include <stdio.h>
#include <stdlib.h>
#include <sys/shm.h>

int main() {
    key_t key = 1234;
    int shmid = shmget(key, 1024, 0666);
    if (shmid<0) { perror("shmget"); exit(1); }

    char *shm = (char*)shmat(shmid, NULL, 0);
    printf("Message: %s\n", shm);
    shmdt(shm);
    shmctl(shmid, IPC_RMID, NULL);
    return 0;
}
