#include <stdio.h>
#include <limits.h>
#include <string.h>

#define MAX 10

typedef struct {
    int pid, at, bt, priority;
    int ft, tat, wt, rem;
} Process;

void display(Process p[], int n) {
    printf("PID\tAT\tBT\tFT\tTAT\tWT\n");
    float tt=0, tw=0;
    for (int i=0;i<n;i++) {
        printf("P%d\t%d\t%d\t%d\t%d\t%d\n",
            p[i].pid,p[i].at,p[i].bt,p[i].ft,p[i].tat,p[i].wt);
        tt+=p[i].tat; tw+=p[i].wt;
    }
    printf("Avg TAT=%.2f  Avg WT=%.2f\n\n",tt/n,tw/n);
}

void fcfs(Process p[], int n) {
    // sort by arrival time
    for (int i=0;i<n-1;i++)
        for (int j=0;j<n-i-1;j++)
            if (p[j].at > p[j+1].at)
                { Process t=p[j]; p[j]=p[j+1]; p[j+1]=t; }
    int time=0;
    for (int i=0;i<n;i++) {
        if (time < p[i].at) time=p[i].at;
        time += p[i].bt;
        p[i].ft=time;
        p[i].tat=p[i].ft-p[i].at;
        p[i].wt=p[i].tat-p[i].bt;
    }
    printf("--- FCFS ---\n");
    display(p,n);
}

void sjf(Process p[], int n) {
    int done[MAX]={0}, time=0, count=0;
    printf("--- SJF ---\n");
    printf("PID\tAT\tBT\tFT\tTAT\tWT\n");
    float tt=0,tw=0;
    while (count<n) {
        int idx=-1, minBT=INT_MAX;
        for (int i=0;i<n;i++)
            if (!done[i] && p[i].at<=time && p[i].bt<minBT)
                { minBT=p[i].bt; idx=i; }
        if (idx==-1) { time++; continue; }
        time+=p[idx].bt;
        p[idx].ft=time;
        p[idx].tat=p[idx].ft-p[idx].at;
        p[idx].wt=p[idx].tat-p[idx].bt;
        done[idx]=1; count++;
        printf("P%d\t%d\t%d\t%d\t%d\t%d\n",
            p[idx].pid,p[idx].at,p[idx].bt,p[idx].ft,p[idx].tat,p[idx].wt);
        tt+=p[idx].tat; tw+=p[idx].wt;
    }
    printf("Avg TAT=%.2f  Avg WT=%.2f\n\n",tt/n,tw/n);
}

void priority_sched(Process p[], int n) {
    int done[MAX]={0}, time=0, count=0;
    printf("--- Priority ---\n");
    printf("PID\tAT\tBT\tPRI\tFT\tTAT\tWT\n");
    float tt=0,tw=0;
    while (count<n) {
        int idx=-1, minP=INT_MAX;
        for (int i=0;i<n;i++)
            if (!done[i] && p[i].at<=time && p[i].priority<minP)
                { minP=p[i].priority; idx=i; }
        if (idx==-1) { time++; continue; }
        time+=p[idx].bt;
        p[idx].ft=time;
        p[idx].tat=p[idx].ft-p[idx].at;
        p[idx].wt=p[idx].tat-p[idx].bt;
        done[idx]=1; count++;
        printf("P%d\t%d\t%d\t%d\t%d\t%d\t%d\n",
            p[idx].pid,p[idx].at,p[idx].bt,p[idx].priority,
            p[idx].ft,p[idx].tat,p[idx].wt);
        tt+=p[idx].tat; tw+=p[idx].wt;
    }
    printf("Avg TAT=%.2f  Avg WT=%.2f\n\n",tt/n,tw/n);
}

void round_robin(Process p[], int n, int tq) {
    int rem[MAX], time=0;
    for (int i=0;i<n;i++) rem[i]=p[i].bt;
    printf("--- Round Robin (TQ=%d) ---\n",tq);
    int finished=0;
    while (finished<n) {
        finished=0;
        for (int i=0;i<n;i++) {
            if (rem[i]>0) {
                int exec = rem[i]>tq ? tq : rem[i];
                time+=exec; rem[i]-=exec;
                if (rem[i]==0) {
                    p[i].ft=time;
                    p[i].tat=p[i].ft-p[i].at;
                    p[i].wt=p[i].tat-p[i].bt;
                }
            }
            if (rem[i]==0) finished++;
        }
    }
    display(p,n);
}

int main() {
    Process p[]={{1,0,5,2},{2,1,3,1},{3,2,8,3},{4,3,6,2}};
    Process p2[MAX];
    int n=4;

    memcpy(p2,p,n*sizeof(Process)); fcfs(p2,n);
    memcpy(p2,p,n*sizeof(Process)); sjf(p2,n);
    memcpy(p2,p,n*sizeof(Process)); priority_sched(p2,n);
    memcpy(p2,p,n*sizeof(Process)); round_robin(p2,n,2);
    return 0;
}
